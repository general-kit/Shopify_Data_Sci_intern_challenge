Select Sum (Quantity), Shippers.ShipperName
from OrderDetails

Join Orders
ON Orders.OrderID = OrderDetails.OrderID

JOIN Shippers
ON Shippers.ShipperID = Orders.ShipperID
where Shippers.Shippername = 'Speedy Express'

/*Speedy Express Total: 3575*/

Select  Employees.LastName, MAX(Quantity)
from OrderDetails

JOIN Orders
ON OrderDetails.OrderID = Orders.OrderID

JOIN Employees
ON Orders.EmployeeID = Employees.EmployeeID

/* Last Name of Employee with most orders: Fuller*/

Select Orders.CustomerID, MAX(OrderDetails.Quantity), Customers.Country, Products.ProductName

From Orders

JOIN Customers
ON Orders.CustomerID = Customers.CustomerID

JOIN OrderDetails
ON Orders.OrderID = OrderDetails.OrderID

JOIN Products
ON Products.ProductID = OrderDetails.ProductID

where Country like 'Germany'

/* Product ordered the most in Germany: Steeleye Stout */
